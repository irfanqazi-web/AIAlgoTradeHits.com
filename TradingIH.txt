        function analyzeStock() {
            // Get input values
            const ticker = document.getElementById('ticker').value.toUpperCase();
            const assetType = document.getElementById('assetType').value;
            const revenueGrowth = parseFloat(document.getElementById('revenueGrowth').value);
            const peRatio = parseFloat(document.getElementById('peRatio').value);
            const pegRatio = parseFloat(document.getElementById('pegRatio').value);
            const roe = parseFloat(document.getElementById('roe').value);
            const quickRatio = parseFloat(document.getElementById('quickRatio').value);
            const marketCap = parseFloat(document.getElementById('marketCap').value);
            
            // Validate inputs
            if (!ticker || isNaN(revenueGrowth) || isNaN(peRatio) || isNaN(pegRatio) || isNaN(roe) || isNaN(quickRatio)) {
                alert('Please fill in all required fields with valid numbers');
                return;
            }
            
            // Perform analysis based on the flowchart criteria
            let score = 0;
            let maxScore = 100;
            const analysisResults = [];
            
            // 1. Revenue Growth Check (20 points)
            if (revenueGrowth >= 10) {
                score += 20;
                analysisResults.push({
                    pass: true,
                    icon: 'üìà',
                    title: 'Revenue Growth',
                    value: ${revenueGrowth}% annually - Strong growth momentum,
                    weight: 20
                });
            } else {
                analysisResults.push({
                    pass: false,
                    icon: 'üìâ',
                    title: 'Revenue Growth',
                    value: ${revenueGrowth}% annually - Below 10% threshold,
                    weight: 20
                });
            }
            
            // 2. P/E Ratio Check (20 points)
            if (peRatio < 25) {
                score += 20;
                analysisResults.push({
                    pass: true,
                    icon: 'üí∞',
                    title: 'P/E Ratio',
                    value: ${peRatio} - Reasonably valued,
                    weight: 20
                });
            } else {
                const overvalued = ((peRatio - 25) / 25 * 100).toFixed(1);
                analysisResults.push({
                    pass: false,
                    icon: '‚ö†',
                    title: 'P/E Ratio',
                    value: ${peRatio} - Potentially overvalued by ${overvalued}%,
                    weight: 20
                });
            }
            
            // 3. PEG Ratio Check (20 points)
            if (pegRatio < 2) {
                score += 20;
                analysisResults.push({
                    pass: true,
                    icon: 'üöÄ',
                    title: 'PEG Ratio',
                    value: ${pegRatio} - Good growth at reasonable price,
                    weight: 20
                });
            } else {
                analysisResults.push({
                    pass: false,
                    icon: 'üêå',
                    title: 'PEG Ratio',
                    value: ${pegRatio} - Low profit growth relative to valuation,
                    weight: 20
                });
            }
            
            // 4. ROE Check (20 points)
            if (roe > 5) {
                score += 20;
                analysisResults.push({
                    pass: true,
                    icon: 'üí™',
                    title: 'Return on Equity',
                    value: ${roe}% average - Strong profitability,
                    weight: 20
                });
            } else {
                analysisResults.push({
                    pass: false,
                    icon: 'üò∞',
                    title: 'Return on Equity',
                    value: ${roe}% average - Weak profitability,
                    weight: 20
                });
            }
            
            // 5. Quick Ratio Check (20 points)
            if (quickRatio > 1.5) {
                score += 20;
                analysisResults.push({
                    pass: true,
                    icon: 'üíß',
                    title: 'Quick Ratio',
                    value: ${quickRatio} - Excellent liquidity,
                    weight: 20
                });
            } else {
                analysisResults.push({
                    pass: false,
                    icon: 'üö®',
                    title: 'Quick Ratio',
                    value: ${quickRatio} - Potential liquidity concerns,
                    weight: 20
                });
            }
            
            // Calculate time horizon scores with different weightings
            const volatilityFactor = assetType === 'crypto' ? 1.5 : 1.0;
            const sizeFactor = marketCap > 100 ? 0.9 : marketCap > 10 ? 1.0 : 1.2;
            
            // Day score: more volatile, less predictable
            const dayScore = Math.max(0, Math.min(100, 
                score * 0.7 + (Math.random() * 30 - 15) * volatilityFactor
            ));
            
            // Week score: slight improvement in predictability
            const weekScore = Math.max(0, Math.min(100, 
                score * 0.85 + (Math.random() * 20 - 10) * volatilityFactor
            ));
            
            // Month score: fundamentals start to matter more
            const monthScore = Math.max(0, Math.min(100, 
                score * 0.95 + (Math.random() * 15 - 7.5) * sizeFactor
            ));
            
            // Quarter score: fundamentals dominate
            const quarterScore = Math.max(0, Math.min(100, 
                score * 1.0 + (Math.random() * 10 - 5) * sizeFactor
            ));
            
            // Display results
            document.getElementById('results').style.display = 'block';
            document.getElementById('overallScore').textContent = score;
            
            // Overall recommendation
            let recommendation = '';
            if (score >= 80) {
                recommendation = 'üü¢ STRONG BUY';
                document.getElementById('overallScore').style.color = '#10b981';
            } else if (score >= 60) {
                recommendation = 'üü¢ BUY';
                document.getElementById('overallScore').style.color = '#34d399';
            } else if (score >= 40) {
                recommendation = 'üü° HOLD';
                document.getElementById('overallScore').style.color = '#fbbf24';
            } else if (score >= 20) {
                recommendation = 'üî¥ SELL';
                document.getElementById('overallScore').style.color = '#f87171';
            } else {
                recommendation = 'üî¥ STRONG SELL';
                document.getElementById('overallScore').style.color = '#dc2626';
            }
            document.getElementById('overallRecommendation').textContent = recommendation;
            
            // Update timeline scores
            updateTimelineScore('day', dayScore);
            updateTimelineScore('week', weekScore);
            updateTimelineScore('month', monthScore);
            updateTimelineScore('quarter', quarterScore);
            
            // Display detailed analysis
            let analysisHTML = '';
            analysisResults.forEach(result => {
                const statusClass = result.pass ? 'pass' : 'fail';
                const badgeClass = result.pass ? 'status-pass' : 'status-fail';
                const badgeText = result.pass ? '‚úì PASS' : '‚úó FAIL';
                
                analysisHTML += `
                    <div class="analysis-item ${statusClass}">
                        <div class="analysis-icon">${result.icon}</div>
                        <div class="analysis-content">
                            <div class="analysis-title">${result.title}</div>
                            <div class="analysis-value">${result.value}</div>
                        </div>
                        <span class="status-badge ${badgeClass}">${badgeText}</span>
                    </div>
                `;
            });
            document.getElementById('analysisDetails').innerHTML = analysisHTML;
            
            // Update methodology
            document.getElementById('methodology').textContent = 
                `Analysis for ${ticker} (${assetType}): This score is calculated using a weighted system based on 5 key financial metrics. ` +
                `Each metric contributes 20 points to the total score. Time horizon predictions adjust for ${assetType === 'crypto' ? 'high cryptocurrency volatility' : 'market volatility'} ` +
                and company size (Market Cap: $${marketCap}B). Short-term predictions are less reliable than long-term fundamental analysis.;
            
            // Scroll to results
            document.getElementById('results').scrollIntoView({ behavior: 'smooth', block: 'nearest' });
        }
        
        function updateTimelineScore(period, score) {
            const roundedScore = Math.round(score);
            document.getElementById(${period}Score).textContent = roundedScore;
            
            const recElement = document.getElementById(${period}Rec);
            let recText = '';
            let recClass = '';
            
            if (roundedScore >= 80) {
                recText = 'STRONG BUY';
                recClass = 'recommendation-strong-buy';
            } else if (roundedScore >= 65) {
                recText = 'BUY';
                recClass = 'recommendation-buy';
            } else if (roundedScore >= 45) {
                recText = 'HOLD';
                recClass = 'recommendation-hold';
            } else if (roundedScore >= 25) {
                recText = 'SELL';
                recClass = 'recommendation-sell';
            } else {
                recText = 'STRONG SELL';
                recClass = 'recommendation-strong-sell';
            }
            
            recElement.textContent = recText;
            recElement.className = timeline-recommendation ${recClass};
            
            // Color the score
            const scoreElement = document.getElementById(${period}Score);
            if (roundedScore >= 70) {
                scoreElement.style.color = '#10b981';
            } else if (roundedScore >= 50) {
                scoreElement.style.color = '#fbbf24';
            } else {
                scoreElement.style.color = '#dc2626';
            }
        }
