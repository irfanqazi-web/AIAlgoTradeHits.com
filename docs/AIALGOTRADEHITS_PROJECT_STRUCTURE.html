8. Deployment Commands

Deploy Cloud Function

cd cloud_functions/bulletproof
gcloud functions deploy bulletproof-fetcher \
  --gen2 --runtime=python312 --region=us-central1 \
  --entry-point=bulletproof_fetch --trigger-http \
  --allow-unauthenticated --memory=4096MB \
  --timeout=540s --project=aialgotradehits

Deploy Dashboard to Cloud Run

cd dashboard/stock-price-app
gcloud run deploy trading-app \
  --source . --platform managed \
  --region us-central1 --allow-unauthenticated \
  --port 8080 --project aialgotradehits
